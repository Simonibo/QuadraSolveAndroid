    public void displayErrorDialog(String message, final int responseID) {
        dismissed = false;
        Log.d("Error", message);
        final Dialog d_err = new Dialog(MainActivity.this);
        d_err.setContentView(R.layout.error_dialog);
        d_err.setTitle("Error");
        TextView error = (TextView) d_err.findViewById(R.id.error);
        error.setText(message);

        Button dialogButton = (Button) d_err.findViewById(R.id.error_ok);
        dialogButton.setOnClickListener(new View.OnClickListener() {
            @Override
            public void onClick(View v) {
                d_err.dismiss();
                onDismiss(responseID);
            }
        });
        d_err.show();
    }

    public void onDismiss(int responseID) {
        switch(responseID) {
            case 1:
                GraphView.roots = roots;
                GraphView.x1 = x1;
                GraphView.x2 = x2;
                GraphView.a = Double.parseDouble(aval.getText().toString());
                GraphView.b = Double.parseDouble(bval.getText().toString());
                GraphView.c = Double.parseDouble(cval.getText().toString());
                Intent intent = new Intent(this, Graph.class);
                startActivity(intent);
                break;
        }
    }
